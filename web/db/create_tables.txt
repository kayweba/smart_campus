create table if not exists Rooms
(
    sdescription text,
    nnumber integer,
    id integer primary key autoincrement
); 

create table if not exists ElectricalSensorType 
(
    sdescription text,
    id integer primary key autoincrement
);

create table ElectricalSensors
(
    sname text,
    nstate integer,
    rvalue real,
    id integer primary key autoincrement,
    id_type integer,
    id_room integer,
    -- Прописываем внешние ключи
    foreign key (id_type) references ElectricalSensorType(id),
    foreign key (id_room) references Rooms(id)
);
